//insertion sort
(SBM)
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int a[7] = {1,5,9,7,3,4,6};
    for(int i=0;i<6;i++){
        if(a[i+1]<a[i]){
            int temp = i;
            while(true){
                if(a[i+1]<a[i]){swap(a[i],a[i+1]);i--;}
                else break;
            }
            i=temp;
        }
    } 

    cout<<"ans: ";
    for(int i=0;i<7;i++)
    cout<<a[i]<<" "
;
    return 0;
}
