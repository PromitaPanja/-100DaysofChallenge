Hackerearth contest - Amex/American Mackethon 2021
(SBM)

You are given an array with length n consisting n integers a1, a2,...,an.
Task-
Determine a subsequence that contains the minimum number of distinct numbers among all possible subsequence of array a which has exactly k length.
input:
n = 10
k = 6
a = [1,3,2,4,3,4,5,3,4,3]
output:
2
Approach:
Subsequence of a with length k is [3,4,3,4,3,4] with two distinct no.s, which is minimum possible. hence ans is 2.

Solution ->
#include<bits/stdc++.h>
using namespace std;
int solve(int n,int k, vector<int> a){
vector<int> c;
map<int,int> b;
for(int i=0;i<n;i++){
b[a[i]]++;
}
map<int,int>::iterator it;
for(it=b.begin();it!=b.end();it++){
c.push_back(it->second);
}
sort(c.begin(),c.end(),greater<int>());

int l = c.size();
int ans = 0;
for(int i=0;i<l;i++){
if(c[i]<=k && k>0){
k-=c[i];
ans++;
}
if(k<=0){
break;}
}
return ans;
}

int main(){
int n,k;
cin>>n>>k;
vector<int> a;
for(auto i:a){
cin>>i;
}
int ans = solve(n,k,a);
cout<<ans;
return 0;
}
